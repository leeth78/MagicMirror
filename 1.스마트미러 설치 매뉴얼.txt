pcm.!default{
  type asym
  playback.pcm{
    type hw
    card 0
  }
  capture.pcm{
    type plug
    slave.pcm "hw:1, 0"
  }
}
ctl.!default{
  type hw
  card 0
}
---------------------------------------------

pcm.!default{
    type asym
    capture.pcm "mic"
    playback.pcm "speaker"
    }
pcm.mic{
    type plug
    slave{
        pcm "hw:1,0"
        }
    }
pcm.speaker{
    type plug
    slave{
        pcm "hw:1,0"
        }
    }
